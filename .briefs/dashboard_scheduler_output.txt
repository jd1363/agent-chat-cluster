    def _post_scheduler_start(self, body):
        """Start the scheduler daemon if not already running."""
        global _SCHEDULER_PROC

        if _SCHEDULER_PROC is not None and _SCHEDULER_PROC.poll() is None:
            self._json({'ok': False, 'error': 'scheduler already running'}, status=409)
            return

        interval = body.get('interval', 10)
        execute_real = body.get('executeReal', False)

        # Build the command: always passes --loop with the interval
        cmd = [sys.executable, str(SCRIPTS_DIR / 'scheduler_tick.py'), '--loop', str(interval)]
        if execute_real:
            cmd.append('--executeReal')

        try:
            proc = subprocess.Popen(
                cmd,
                cwd=str(PROJECT_ROOT),
                stdout=subprocess.DEVNULL,
                stderr=subprocess.DEVNULL,
                env=SUB_ENV,
            )
            _SCHEDULER_PROC = proc
            self._json({'ok': True, 'pid': proc.pid})
        except Exception as e:
            _SCHEDULER_PROC = None
            self._json({'ok': False, 'error': f'failed to start scheduler: {e}'}, status=500)

    def _post_scheduler_stop(self, body):
        """Stop the running scheduler daemon."""
        global _SCHEDULER_PROC

        if _SCHEDULER_PROC is None:
            self._json({'ok': False, 'error': 'scheduler not running'}, status=400)
            return

        proc = _SCHEDULER_PROC
        if proc.poll() is not None:
            # Process already exited
            _SCHEDULER_PROC = None
            self._json({'ok': True, 'message': 'scheduler already stopped'})
            return

        # Terminate gracefully, then force if needed
        proc.terminate()
        try:
            proc.wait(timeout=3)
        except subprocess.TimeoutExpired:
            proc.kill()
            proc.wait()
        _SCHEDULER_PROC = None
        self._json({'ok': True, 'message': 'scheduler stopped'})

    def _get_scheduler_status(self):
        """Return current scheduler daemon status."""
        global _SCHEDULER_PROC

        if _SCHEDULER_PROC is None:
            self._json({'ok': True, 'running': False})
            return

        proc = _SCHEDULER_PROC
        if proc.poll() is None:
            # Still running
            self._json({
                'ok': True,
                'running': True,
                'pid': proc.pid
            })
        else:
            # Process exited on its own
            self._json({
                'ok': True,
                'running': False,
                'exitCode': proc.returncode
            })
            # Optionally clear the global so next start works
            global _SCHEDULER_PROC
            _SCHEDULER_PROC = None
